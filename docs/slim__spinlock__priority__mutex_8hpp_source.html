98 void lock(Priority_t const priority = 0){
99
100 control_block_t localCtrl = ctrl_.load();
101
102 // try to increment the waiters counter in this priority
103 for(;;){
104 // if we risk an overflow lets spin instead
105 if (localCtrl.priority_[priority] == uint8_t(-1)){
106 localCtrl = ctrl_.load();
107 continue;
108 }
109 if (ctrl_.compare_exchange_weak(localCtrl, localCtrl.increasePriority(priority))){
110 break;
111 }
112 }
113 // if we can proceed we do and decrement the counter otherwise we wait on an atomic_flag
114 for(;;){
115 localCtrl = ctrl_.load();
116 if (!lockOwned_(localCtrl.owned_) && priority <= localCtrl.owned_ && ctrl_.compare_exchange_strong(localCtrl, localCtrl.setOwned())){
117 break;
118 }
119 }
